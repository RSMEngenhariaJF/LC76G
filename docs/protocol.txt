// pData is the data array whose checksum needs to be calculated:

unsigned char Ql_Check_XOR(const unsigned char *pData, unsigned int Length)
{

      unsigned char result = 0;
      unsigned int i = 0;

      if((NULL == pData) || (Length < 1))
      {

            return 0;
      }
      for(i = 0; i < Length; i++)
      {

            result ^= *(pData + i);
      }

      return result;
}
